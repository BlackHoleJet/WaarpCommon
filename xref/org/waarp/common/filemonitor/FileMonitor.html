213 	public void start() {
214 		if (timer == null) {
215 			timer = new HashedWheelTimer(
216 						new WaarpThreadFactory("TimerFileMonitor"),
217 						100, TimeUnit.MILLISECONDS, 8);
218 			timer.newTimeout(new FileMonitorTimerTask(this), elapseTime, TimeUnit.MILLISECONDS);
219 			future = new WaarpFuture(true);
220 			internalfuture = new WaarpFuture(true);
221 		}// else already started
222 	}

224 	public void stop() {
225 		if (timer != null) {
226 			timer.stop();
227 		}
228 		if (internalfuture != null) {
229 			internalfuture.awaitUninterruptibly();
230 			internalfuture = null;
231 		}
232 		timer = null;
233 		if (future != null) {
234 			future.setSuccess();
235 		}
236 	}

264 	/**
265 	 * Check Files
266 	 * @return False to stop
267 	 */
268 	protected boolean checkFiles() {
269 		boolean fileItemsChanged = false;
270 		if (stopFile.exists()) {
271 			return false;
272 		}
273 		for (File directory : directories) {
274 			fileItemsChanged = checkOneDir(fileItemsChanged, directory);
275 		}
276 		// now check that all existing items are still valid
277 		List<FileItem> todel = new LinkedList<FileItem>();
278 		for (FileItem item : fileItems.values()) {
279 			if (item.file.isFile()) {
280 				continue;
281 			}
282 			todel.add(item);
283 		}
284 		// remove invalid files
285 		for (FileItem fileItem : todel) {
286 			String name = AbstractDir.normalizePath(fileItem.file.getAbsolutePath());
287 			fileItems.remove(name);
288 			toUse.remove(fileItem);
289 			if (commandRemovedFile != null) {
290 				commandRemovedFile.run(fileItem.file);
291 			}
292 			fileItem.file = null;
293 			fileItem.hash = null;
294 			fileItem = null;
295 			fileItemsChanged = true;
296 		}
297 		if (fileItemsChanged) {
298 			this.saveStatus();
299 		}
300 		if (commandCheckIteration != null) {
301 			commandCheckIteration.run(null);
302 		}
303 		return true;
304 	}

306 	/**
307 	 * @param fileItemsChanged
308 	 * @param directory
309 	 * @return True if one file at least has changed
310 	 */
311 	protected boolean checkOneDir(boolean fileItemsChanged, File directory) {
312 		File [] files = directory.listFiles(filter);
313 		for (File file : files) {
314 			if (file.isDirectory()) {
315 				continue;
316 			}
317 			String name = AbstractDir.normalizePath(file.getAbsolutePath());
318 			FileItem fileItem = fileItems.get(name);
319 			if (fileItem == null) {
320 				// never seen until now
321 				fileItems.put(name, new FileItem(file));
322 				fileItemsChanged = true;
323 				continue;
324 			}
325 			if (fileItem.used) {
326 				// already used so ignore
327 				continue;
328 			}
329 			long lastTimeModified = fileItem.file.lastModified();
330 			if (lastTimeModified != fileItem.lastTime) {
331 				// changed or second time check
332 				fileItem.lastTime = lastTimeModified;
333 				fileItemsChanged = true;
334 				continue;
335 			}
336 			// now check Hash or third time
337 			try {
338 				byte [] hash = FilesystemBasedDigest.getHash(fileItem.file, true, digest);
339 				if (hash == null || fileItem.hash == null) {
340 					fileItem.hash = hash;
341 					fileItemsChanged = true;
342 					continue;
343 				}
344 				if (! Arrays.equals(hash, fileItem.hash)) {
345 					fileItem.hash = hash;
346 					fileItemsChanged = true;
347 					continue;
348 				}
349 				// now time and hash are the same so act on it
350 				fileItem.timeUsed = System.currentTimeMillis();
351 				if (commandValidFile != null) {
352 					if (commandValidFile.run(fileItem.file)) {
353 						fileItem.used = true;
354 						fileItem.hash = null;
355 					}
356 				} else {
357 					toUse.add(fileItem);
358 				}
359 				fileItemsChanged = true;
360 			} catch (IOException e) {
361 				continue;
362 			}
363 		}
364 		if (scanSubDir) {
365 			files = directory.listFiles();
366 			for (File file : files) {
367 				if (file.isDirectory()) {
368 					fileItemsChanged = checkOneDir(fileItemsChanged, file);
369 				}
370 			}
371 		}
372 		return fileItemsChanged;
373 	}

389 		public void run(Timeout timeout) throws Exception {
390 			if (fileMonitor.checkFiles()) {
391 				fileMonitor.timer.newTimeout(this, fileMonitor.elapseTime, TimeUnit.MILLISECONDS);
392 			} else {
393 				fileMonitor.internalfuture.setSuccess();
394 			}
395 		}

469 	public static void main(String[] args) {
470     	if (args.length < 3) {
471     		System.err.println("Need a statusfile, a stopfile and a directory to test");
472     		return;
473     	}
474     	File file = new File(args[0]);
475     	if (file.exists() && ! file.isFile()) {
476     		System.err.println("Not a correct status file");
477     		return;
478     	}
479     	File stopfile = new File(args[1]);
480     	if (file.exists() && ! file.isFile()) {
481     		System.err.println("Not a correct stop file");
482     		return;
483     	}
484     	File dir = new File(args[2]);
485     	if (! dir.isDirectory()) {
486     		System.err.println("Not a directory");
487     		return;
488     	}
489     	FileMonitor monitor = new FileMonitor("test", file, stopfile, dir, null, 0, 
490     			new RegexFileFilter(RegexFileFilter.REGEX_XML_EXTENSION), 
491     			false, new FileMonitorCommand() {
492 			public boolean run(File file) {
493 				System.out.println("File New: "+file.getAbsolutePath());
494 				return true;
495 			}
496 		}, new FileMonitorCommand() {
497 			public boolean run(File file) {
498 				System.err.println("File Del: "+file.getAbsolutePath());
499 				return true;
500 			}
501 		}, new FileMonitorCommand() {
502 			public boolean run(File unused) {
503 				System.err.println("Check done");
504 				return true;
505 			}
506 		});
507     	monitor.start();
508     	monitor.waitForStopFile();
509 	}
