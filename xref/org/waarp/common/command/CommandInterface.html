21  package org.waarp.common.command;
22  
23  import org.waarp.common.command.exception.CommandAbstractException;
24  import org.waarp.common.exception.InvalidArgumentException;
25  import org.waarp.common.file.SessionInterface;
26  
27  
28  /**
29   * Interface for Command
30   *
31   * @author Frederic Bregier
32   *
33   */
34  public interface CommandInterface {
35      /**
36       * Set the Command from the args
37       *
38       * @param session
39       * @param command
40       * @param arg
41       * @param code
42       */
43      public void setArgs(SessionInterface session, String command, String arg,
44              @SuppressWarnings("rawtypes") Enum code);
45  
46      /**
47       * Execute the command. This execution must set the replyCode in the session
48       * to a correct value before returning.
49       *
50       * @exception CommandAbstractException
51       *                in case of an FTP Error occurs
52       */
53      abstract public void exec() throws CommandAbstractException;
54  
55      /**
56       * This function is intend to allow to force USER->PASS->ACCT->CDW for
57       * instance
58       *
59       * @param extraNextCommand
60       *            the extraNextCommand to set
61       */
62      public void setExtraNextCommand(@SuppressWarnings("rawtypes") Enum extraNextCommand);
63  
64      /**
65       * This function is called when a new command is received to check if this
66       * new command is positive according to the previous command and status.
67       *
68       * @param newCommand
69       * @return True if this new command is OK, else False
70       */
71      public abstract boolean isNextCommandValid(CommandInterface newCommand);
72  
73      /**
74       * @return the object
75       */
76      public Object getObject();
77  
78      /**
79       * @param object
80       *            the object to set
81       */
82      public void setObject(Object object);
83  
84      /**
85       * @return the arg
86       */
87      public String getArg();
88  
89      /**
90       *
91       * @return the list of arguments
92       */
93      public String[] getArgs();
94  
95      /**
96       * Get an integer value from argument
97       *
98       * @param argx
99       * @return the integer
100      * @throws InvalidArgumentException
101      *             if the argument is not an integer
102      */
103     public int getValue(String argx) throws InvalidArgumentException;
104 
105     /**
106      * @return the command
107      */
108     public String getCommand();
109 
110     /**
111      * Does this command has an argument
112      *
113      * @return True if it has an argument
114      */
115     public boolean hasArg();
116 
117     /**
118      *
119      * @return the current SessionInterface
120      */
121     public SessionInterface getSession();
122 
123     // some helpful functions
124     /**
125      * Set the previous command as the new current command (used after a
126      * incorrect sequence of commands or unknown command). Also clear the
127      * Restart object.
128      *
129      */
130     public void invalidCurrentCommand();
131 
132     /**
133      *
134      * @return The GgCommandCode associated with this command
135      */
136     public Enum<?> getCode();
137 }
