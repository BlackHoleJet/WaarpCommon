39  /**
40   * UUID Generator (also Global UUID Generator) <br>
41   * <br>
42   * Inspired from com.groupon locality-uuid which used combination of internal counter value - process id - 
43   * fragment of MAC address and Timestamp. see https://github.com/groupon/locality-uuid.java <br>
44   * <br>
45   * But force sequence and take care of errors and improves some performance issues
46   *  
47   * @author "Frederic Bregier"
48   *
49   */
50  public final class UUID {
51  	
52      private static final int KEYSIZE			= 18;
53      private static final int KEYB64SIZE			= 24;
54      private static final int KEYB16SIZE			= KEYSIZE*2;
55      /**
56       * Random Generator 
57       */
58      private static final ThreadLocalRandom RANDOM			= ThreadLocalRandom.current();
59      /**
60       * So MAX value on 2 bytes
61       */
62      private static final int MAX_PID			= 65536;
63      /**
64       * Version to store (to check correctness if future algorithm)
65       */
66      private static final char VERSION			= 'd';
67      /**
68       * VERSION_DEC
69       */
70      private static final int VERSION_DEC = Hexa.asByte(VERSION, '0');
71  
72      private static final Pattern MACHINE_ID_PATTERN = Pattern.compile("^(?:[0-9a-fA-F][:-]?){6,8}$");
73      private static final int MACHINE_ID_LEN = 6;
74  
75  	/**
76  	 * 2 bytes value maximum
77  	 */
78      private static final int JVMPID = jvmProcessId();
79      /**
80       * Try to get Mac Address but could be also changed dynamically
81       */
82      private static byte[] MAC = macAddress();
83      /**
84       * Counter part
85       */
86      private static final AtomicInteger COUNTER = new AtomicInteger(RANDOM.nextInt());

88      /**
89       * real UUID
90       */
91      private final byte[] uuid;

94      /**
95       * Up to the 6 first bytes will be used. If Null or less than 6 bytes, extra bytes will
96       * be randomly generated.
97       * 
98       * @param mac the MAC address in byte format (up to the 6 first bytes will be used)
99       */
100     public static synchronized void setMAC(final byte []mac) {
101     	if (mac == null) {
102         	MAC = getRandom(MACHINE_ID_LEN);
103     	} else {
104 	    	MAC = Arrays.copyOf(mac, MACHINE_ID_LEN);
105 	    	for (int i = mac.length; i < MACHINE_ID_LEN; i++) {
106 	    		MAC[i] = (byte) RANDOM.nextInt(256);
107 	    	}
108     	}
109     }

111     /**
112      * Constructor that generates a new UUID using the current process id, MAC address, and timestamp
113      */
114     public UUID() {
115     	final long time = System.currentTimeMillis();
116         uuid = new byte[KEYSIZE];
117 
118         // atomically
119         final int count = COUNTER.incrementAndGet();
120 
121         // switch the order of the count in 3 bit segments and place into uuid
122         uuid[0] = (byte) (((count & 0x0F) << 4) | ((count & 0xF0) >> 4));
123         uuid[1] = (byte) (((count & 0xF00) >> 4) | ((count & 0xF000) >> 12));
124         uuid[2] = (byte) (((count & 0xF0000) >> 12) | ((count & 0xF00000) >> 20));
125         //uuid[3] = (byte) (((count & 0xF000000) >> 20) | ((count & 0xF0000000) >> 28));
126 
127         // copy pid to uuid
128         uuid[3]  = (byte) (JVMPID >> 8);
129         uuid[4]  = (byte) (JVMPID);
130 
131 
132      // place UUID version (hex 'c') in first four bits and piece of MAC in
133         // the second four bits
134         uuid[5]  = (byte) (VERSION_DEC | (0x0F & MAC[0]));
135         // copy rest of mac address into uuid
136         uuid[6]  = MAC[1];
137         uuid[7]  = MAC[2];
138         uuid[8]  = MAC[3];
139         uuid[9]  = MAC[4];
140         uuid[10]  = MAC[5];
141 
142         // copy timestamp into uuid (up to 48 bits so up to 2 200 000 years after Time 0)
143         uuid[11] = (byte) (time >> 48);
144         uuid[12] = (byte) (time >> 40);
145         uuid[13] = (byte) (time >> 32);
146         uuid[14] = (byte) (time >> 24);
147         uuid[15] = (byte) (time >> 16);
148         uuid[16] = (byte) (time >> 8);
149         uuid[17] = (byte) (time);
150     }

154      * @param bytes UUID content

156     public UUID(final byte[] bytes) {
157         if (bytes.length != KEYSIZE)
158             throw new RuntimeException("Attempted to parse malformed UUID: " + Arrays.toString(bytes));
159 
160         uuid = Arrays.copyOf(bytes, KEYSIZE);
161     }

163     public UUID(final String idsource) {
164     	final String id = idsource.trim();
165 
166     	int len = id.length();
167         if (len == KEYB16SIZE) {
168         	// HEXA
169         	uuid = Hexa.fromHex(id);
170         } else if (len == KEYB64SIZE || len == KEYB64SIZE+1) {
171         	// BASE64
172     		try {
173 				uuid = Base64.decode(id, Base64.URL_SAFE|Base64.DONT_GUNZIP);
174 			} catch (IOException e) {
175 	            throw new RuntimeException("Attempted to parse malformed UUID: " + id, e);
176 			}
177        	} else {
178        		throw new RuntimeException("Attempted to parse malformed UUID: ("+len+") " + id);
179        	}
180     }

215     /**
216      * extract process id from raw UUID bytes and return as int
217      * @return id of process that generated the UUID, or -1 for unrecognized format
218      */
219     public int getProcessId() {
220         if (getVersion() != VERSION)
221             return -1;
222 
223         return ((uuid[3] & 0xFF) << 8) | (uuid[4] & 0xFF);
224     }

226     /**
227      * extract timestamp from raw UUID bytes and return as int
228      * @return millisecond UTC timestamp from generation of the UUID, or -1 for unrecognized format
229      */
230     public long getTimestamp() {
231     	if (getVersion() != VERSION)
232             return -1;
233 
234         long time;
235         time  = ((long)uuid[11] & 0xFF) << 48;
236         time |= ((long)uuid[12] & 0xFF) << 40;
237         time |= ((long)uuid[13] & 0xFF) << 32;
238         time |= ((long)uuid[14] & 0xFF) << 24;
239         time |= ((long)uuid[15] & 0xFF) << 16;
240         time |= ((long)uuid[16] & 0xFF) << 8;
241         time |= ((long)uuid[17] & 0xFF);
242         return time;
243     }

245     /**
246      * extract MAC address fragment from raw UUID bytes, setting missing values to 0,
247      * thus the first half byte will be 0, followed by 7 and half bytes
248      * of the active MAC address when the UUID was generated
249      * @return byte array of UUID fragment, or null for unrecognized format
250      */
251     public byte[] getMacFragment() {
252         if (getVersion() != VERSION)
253             return null;
254 
255         final byte[] x = new byte[6];
256 
257         x[0] = (byte) (uuid[5] & 0x0F);
258         x[1] = uuid[6];
259         x[2] = uuid[7];
260         x[3] = uuid[8];
261         x[4] = uuid[9];
262         x[5] = uuid[10];
263 
264         return x;
265     }

332     private static byte[] parseMachineId(String value) {
333         // Strip separators.
334         value = value.replaceAll("[:-]", "");
335 
336         byte[] machineId = new byte[MACHINE_ID_LEN];
337         for (int i = 0; i < value.length() && i < MACHINE_ID_LEN; i += 2) {
338             machineId[i] = (byte) Integer.parseInt(value.substring(i, i + 2), 16);
339         }
340 
341         return machineId;
342     }

344     private static byte[] defaultMachineId() {
345         // Find the best MAC address available.
346         final byte[] NOT_FOUND = { -1 };
347         byte[] bestMacAddr = NOT_FOUND;
348         InetAddress bestInetAddr = null;
349         try {
350             bestInetAddr = InetAddress.getByAddress(new byte[] { 127, 0, 0, 1 });
351         } catch (UnknownHostException e) {
352             // Never happens.
353         	throw new IllegalArgumentException(e);
354         }
355 
356         // Retrieve the list of available network interfaces.
357         Map<NetworkInterface, InetAddress> ifaces = new LinkedHashMap<NetworkInterface, InetAddress>();
358         try {
359             for (Enumeration<NetworkInterface> i = NetworkInterface.getNetworkInterfaces(); i.hasMoreElements();) {
360                 NetworkInterface iface = i.nextElement();
361                 // Use the interface with proper INET addresses only.
362                 Enumeration<InetAddress> addrs = iface.getInetAddresses();
363                 if (addrs.hasMoreElements()) {
364                     InetAddress a = addrs.nextElement();
365                     if (!a.isLoopbackAddress()) {
366                         ifaces.put(iface, a);
367                     }
368                 }
369             }
370         } catch (SocketException e) {
371         }
372 
373         for (Entry<NetworkInterface, InetAddress> entry: ifaces.entrySet()) {
374             NetworkInterface iface = entry.getKey();
375             InetAddress inetAddr = entry.getValue();
376             if (iface.isVirtual()) {
377                 continue;
378             }
379 
380             byte[] macAddr;
381             try {
382                 macAddr = iface.getHardwareAddress();
383             } catch (SocketException e) {
384                 continue;
385             }
386 
387             boolean replace = false;
388             int res = compareAddresses(bestMacAddr, macAddr);
389             if (res < 0) {
390                 // Found a better MAC address.
391                 replace = true;
392             } else if (res == 0) {
393                 // Two MAC addresses are of pretty much same quality.
394                 res = compareAddresses(bestInetAddr, inetAddr);
395                 if (res < 0) {
396                     // Found a MAC address with better INET address.
397                     replace = true;
398                 } else if (res == 0) {
399                     // Cannot tell the difference.  Choose the longer one.
400                     if (bestMacAddr.length < macAddr.length) {
401                         replace = true;
402                     }
403                 }
404             }
405 
406             if (replace) {
407                 bestMacAddr = macAddr;
408                 bestInetAddr = inetAddr;
409             }
410         }
411 
412         if (bestMacAddr == NOT_FOUND) {
413         	bestMacAddr = getRandom(MACHINE_ID_LEN);
414         }
415         return bestMacAddr;
416     }

418     /**
419      * @return positive - current is better, 0 - cannot tell from MAC addr, negative - candidate is better.
420      */
421     private static int compareAddresses(byte[] current, byte[] candidate) {
422         if (candidate == null) {
423             return 1;
424         }
425         // Must be EUI-48 or longer.
426         if (candidate.length < 6) {
427             return 1;
428         }
429         // Must not be filled with only 0 and 1.
430         boolean onlyZeroAndOne = true;
431         for (byte b: candidate) {
432             if (b != 0 && b != 1) {
433                 onlyZeroAndOne = false;
434                 break;
435             }
436         }
437         if (onlyZeroAndOne) {
438             return 1;
439         }
440         // Must not be a multicast address
441         if ((candidate[0] & 1) != 0) {
442             return 1;
443         }
444         // Prefer globally unique address.
445         if ((current[0] & 2) == 0) {
446             if ((candidate[0] & 2) == 0) {
447                 // Both current and candidate are globally unique addresses.
448                 return 0;
449             } else {
450                 // Only current is globally unique.
451                 return 1;
452             }
453         } else {
454             if ((candidate[0] & 2) == 0) {
455                 // Only candidate is globally unique.
456                 return -1;
457             } else {
458                 // Both current and candidate are non-unique.
459                 return 0;
460             }
461         }
462     }

464     /**
465      * @return positive - current is better, 0 - cannot tell, negative - candidate is better
466      */
467     private static int compareAddresses(InetAddress current, InetAddress candidate) {
468         return scoreAddress(current) - scoreAddress(candidate);
469     }
470 
471     private static int scoreAddress(InetAddress addr) {
472         if (addr.isAnyLocalAddress()) {
473             return 0;
474         }
475         if (addr.isMulticastAddress()) {
476             return 1;
477         }
478         if (addr.isLinkLocalAddress()) {
479             return 2;
480         }
481         if (addr.isSiteLocalAddress()) {
482             return 3;
483         }
484 
485         return 4;
486     }

488     // pulled from http://stackoverflow.com/questions/35842/how-can-a-java-program-get-its-own-process-id
489     public static int jvmProcessId() {
490         // Note: may fail in some JVM implementations
491         // something like '<pid>@<hostname>', at least in SUN / Oracle JVMs
492     	try {
493 	        final String jvmName = ManagementFactory.getRuntimeMXBean().getName();
494 	        final int index = jvmName.indexOf('@');
495 	
496 	        if (index < 1) {
497 	        	System.err.println("Could not get JVMPID");
498 	        	return RANDOM.nextInt(MAX_PID);
499 	        }
500 	        try {
501 	            return Integer.parseInt(jvmName.substring(0, index)) % MAX_PID;
502 	        } catch (NumberFormatException e) {
503 	        	System.err.println("Could not get JVMPID");
504 	        	e.printStackTrace();
505 	        	return RANDOM.nextInt(MAX_PID);
506 	        }
507     	} catch (Exception e) {
508         	e.printStackTrace();
509     		return RANDOM.nextInt(MAX_PID);
510     	}
