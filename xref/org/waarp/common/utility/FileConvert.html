50  	private boolean	unix2dos	= false;

52  	private File		tmpDir		= new File(System.getProperty("java.io.tmpdir"));

55  	/**
56  	 * 
57  	 * @param files list of files
58  	 * @param unix2dos True for Unix2Dos, False for Dos2Unix
59  	 * @param recursive True for Recursive from files, else unitary files
60  	 * @param tmpDir if not null, specific tmp directory
61  	 */
62  	public FileConvert(List<File> files, boolean unix2dos, boolean recursive, File tmpDir) {
63  		if (logger == null) {
64  			logger = WaarpInternalLoggerFactory.getLogger(FileConvert.class);
65  		}
66  		this.files = files;
67  		this.unix2dos = unix2dos;
68  		this.recursive = recursive;
69  		if (tmpDir != null) {
70  			this.tmpDir = tmpDir;
71  		}
72  	}
73  	/**
74  	 * Direct call
75  	 * @param args
76  	 */
77  	public static void main(String[] args) {
78  		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
79  		if (logger == null) {
80  			logger = WaarpInternalLoggerFactory.getLogger(FileConvert.class);
81  		}
82  		
83  		ArrayList<File> files = new ArrayList<File>();
84  		boolean unix2dos = false, dos2unix = false, recursive = false;
85  		File tmpDir = null;
86  		for (int i = 0; i < args.length; i++) {
87  			if (Pattern.compile("^-(u|-unix2dos|-unixtodos)$", Pattern.CASE_INSENSITIVE).matcher(args[i]).matches()) {
88  				unix2dos = true;
89  			} else if (Pattern.compile("^-(d|-dos2unix|-dostounix)$", Pattern.CASE_INSENSITIVE).matcher(args[i]).matches()) {
90  				dos2unix = true;
91  			} else if (Pattern.compile("^-(r|-recursive)$", Pattern.CASE_INSENSITIVE).matcher(args[i]).matches()) {
92  				recursive = true;
93  			} else if (Pattern.compile("^-(t|-temporary)$", Pattern.CASE_INSENSITIVE).matcher(args[i]).matches()) {
94  				tmpDir = new File(args[++i]);
95  			} else {
96  				files.add(new File(args[i]));
97  			}
98  		}
99  		if ((unix2dos == true) && (dos2unix == true)) {
100 			syntax();
101 			System.exit(1);
102 		}
103 		if ((unix2dos == false) && (dos2unix == false)) {
104 			syntax();
105 			System.exit(1);
106 		}
107 		FileConvert fileConvert = new FileConvert(files, unix2dos, recursive, tmpDir);
108 		fileConvert.run();
109 	}

111 	/**
112 	 * Use parameters given in instantiation
113 	 */
114 	public void run() {
115 		if (files == null) {
116 			return;
117 		}
118 		for (Iterator<File> i = files.iterator(); i.hasNext();) {
119 			File file = i.next();
120 			if (file.isDirectory()) {
121 				if (recursive) {
122 					recursive(file);
123 				}
124 			} else {
125 				convert(file, unix2dos);
126 			}
127 		}
128 	}

130 	private void recursive(File directory) {
131 		File[] files = directory.listFiles();
132 		for (int i = 0; i < files.length; i++) {
133 			if (files[i].isDirectory()) {
134 				recursive(files[i]);
135 			} else {
136 				convert(files[i], unix2dos);
137 			}
138 		}
139 	}

141 	private boolean copyFile(File source, File destination) {
142 		FileInputStream in = null;
143 		FileChannel srcChannel = null;
144 		FileOutputStream out = null;
145 		FileChannel dstChannel = null;
146 		try {
147 			in = new FileInputStream(source);
148 			srcChannel = in.getChannel();
149 			out = new FileOutputStream(destination);
150 			dstChannel = out.getChannel();
151 			long src = srcChannel.size();
152 			long dst = dstChannel.transferFrom(srcChannel, 0, src);
153 			return (src == dst);
154 		} catch (IOException e) {
155 			logger.error("FileConvert copy back in error", e);
156 			return false;
157 		} finally {
158 			if (in != null) {
159 				try {
160 					in.close();
161 				} catch (IOException e) {
162 				}
163 			}
164 			if (srcChannel != null) {
165 				try {
166 					srcChannel.close();
167 				} catch (IOException e) {
168 				}
169 			}
170 			if (out != null) {
171 				try {
172 					out.close();
173 				} catch (IOException e) {
174 				}
175 			}
176 			if (dstChannel != null) {
177 				try {
178 					dstChannel.close();
179 				} catch (IOException e) {
180 				}
181 			}
182 		}
183 	}

185 	/**
186 	 * Convert Unix2Dos or Dos2Unix file in place, according to second argument
187 	 * @param input File to change
188 	 * @param unix2dos True for UnixToDos, False for DosToUnix
189 	 * @return True if OK
190 	 */
191 	public boolean convert(File input, boolean unix2dos) {
192 		if (unix2dos) {
193 			logger.info("unix2Dos conversion of '" + input + "'... ");
194 		} else {
195 			logger.info("dos2Unix conversion of '" + input + "'... ");
196 		}
197 		FileInputStream fis = null;
198 		FileOutputStream fos = null;
199 		File tmpFile = null;
200 		try {
201 			fis = new FileInputStream(input);
202 			tmpFile = File.createTempFile("FileConevrt_"+input.getName(), null, tmpDir);
203 			fos = new FileOutputStream(tmpFile);
204 			if (unix2dos) {
205 				byte pb = -1;
206 				byte b = -1;
207 				while ((b = (byte) fis.read()) != -1) {
208 					if ((b == 10) && (pb != 13)) {
209 						fos.write((byte) 13);
210 					}
211 					fos.write(b);
212 					pb = b;
213 				}
214 			} else {
215 				byte b = -1;
216 				byte nb = -1;
217 				while ((b = (byte) fis.read()) != -1) {
218 					if (b == 13) {
219 						nb = (byte) fis.read();
220 						if (nb == -1) {
221 							fos.write(b);
222 						} else {
223 							if (nb == 10) {
224 								fos.write(nb);
225 							} else {
226 								fos.write(b);
227 								fos.write(nb);
228 							}
229 						}
230 					} else {
231 						fos.write(b);
232 					}
233 				}
234 			}			
235 			try {
236 				fis.close();
237 			} catch (IOException e) {
238 			}
239 			fis = null;
240 			try {
241 				fos.close();
242 			} catch (IOException e) {
243 			}
244 			fos = null;
245 			boolean result = copyFile(tmpFile, input);
246 			tmpFile.delete();
247 			tmpFile = null;
248 			if (result) {
249 				logger.info("done.");
250 			} else {
251 				logger.error("FileConvert in error during final copy: "+input+":"+unix2dos);
252 			}
253 			return result;
254 		} catch (FileNotFoundException e) {
255 			logger.error("FileConvert in error: "+input+":"+unix2dos, e);
256 			return false;
257 		} catch (IOException e) {
258 			logger.error("FileConvert in error: "+input+":"+unix2dos, e);
259 			return false;
260 		} finally {
261 			if (tmpFile != null) {
262 				tmpFile.delete();
263 			}
264 			if (fis != null) {
265 				try {
266 					fis.close();
267 				} catch (IOException e) {
268 				}
269 			}
270 			if (fos != null) {
271 				try {
272 					fos.close();
273 				} catch (IOException e) {
274 				}
275 			}
276 		}
277 	}
