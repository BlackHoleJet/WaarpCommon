20  /**
21   * This class handles methods to encrypt and unencrypt messages with any available algorithms in the
22   * JVM.<br>
23   * <b>AES is the best compromise in term of security and efficiency.</b> <br>
24   * Usage:<br>
25   * <ul>
26   * <li>Create a Key object: DynamicKey key = new DynamicKey(size,algo,instance,extension);<br>
27   * As DynamicKey(56, "DES", "DES/ECB/PKCS5Padding", "des") for DES<br>
28   * or through Enum class INSTANCES (minimal supported size) or INSTANCESMAX (maximal supported size)
29   * as DynamicKey(INSTANCESMAX.AES,"aes")</li>
30   * <li>Create a key:
31   * <ul>
32   * <li>Generate: key.generateKey();<br>
33   * The method key.getSecretKeyInBytes() allow getting the key in Bytes.</li>
34   * <li>From an external source: key.setSecretKey(arrayOfBytes);</li>
35   * </ul>
36   * </li>
37   * <li>To encrypt a String in a Base64 format: String myStringCrypt = key.cryptToString(myString);</li>
38   * <li>To unencrypt one string from Base64 format to the original String: String myStringDecrypt =
39   * key.decryptStringInString(myStringCrypte);</li>
40   * </ul>
41   * 
42   * @author frederic bregier
43   * 
44   */
45  public class DynamicKeyObject extends KeyObject {

214 	/**
215 	 * test function
216 	 * 
217 	 * @param plaintext
218 	 * @param size
219 	 * @param algo
220 	 * @throws Exception
221 	 */
222 	private static void test(String plaintext, int size, String algo)
223 			throws Exception {
224 		DynamicKeyObject dyn = new DynamicKeyObject(size, algo, algo, algo);
225 		// Generate a key
226 		dyn.generateKey();
227 		// get the generated key
228 		byte[] secretKey = dyn.getSecretKeyInBytes();
229 		// crypt one text
230 		byte[] ciphertext = dyn.crypt(plaintext);
231 		// print the cipher
232 		// System.out.println("ciphertext = " + dyn.encodeHex(ciphertext));
233 
234 		// Test the set Key
235 		dyn.setSecretKey(secretKey);
236 		// decrypt the cipher
237 		String plaintext2 = dyn.decryptInString(ciphertext);
238 		// print the result
239 		// System.out.println("plaintext2 = " + plaintext2);
240 		if (!plaintext2.equals(plaintext))
241 			System.out.println("Error: plaintext2 != plaintext");
242 
243 		// same on String only
244 		int nb = 100000;
245 		long time1 = System.currentTimeMillis();
246 		for (int i = 0; i < nb; i++) {
247 			String cipherString = dyn.cryptToHex(plaintext);
248 			// System.out.println("cipherString = " + cipherString);
249 			String plaintext3 = dyn.decryptHexInString(cipherString);
250 			// System.out.println("plaintext3 = " + plaintext3);
251 			if (!plaintext3.equals(plaintext))
252 				System.out.println("Error: plaintext3 != plaintext");
253 		}
254 		long time2 = System.currentTimeMillis();
255 		System.out.println(algo + ": Total time: " + (time2 - time1) + " ms, " +
256 				(nb * 1000 / (time2 - time1)) + " crypt or decrypt/s");
257 	}
