61  	@Override
62  	public List<String> initFromList(List<String> keys, String extension) {
63  		LinkedList<String> wrong = new LinkedList<String>();
64  		for (String filename : keys) {
65  			File file = new File(filename);
66  			if (file.canRead()) {
67  				String basename = file.getName();
68  				int lastpos = basename.lastIndexOf(extension);
69  				if (lastpos <= 0) {
70  					wrong.add(filename);
71  					continue;
72  				}
73  				String firstname = basename.substring(0, lastpos - 1);
74  				int len = (int) file.length();
75  				byte[] key = new byte[len];
76  				FileInputStream inputStream = null;
77  				try {
78  					inputStream = new FileInputStream(file);
79  				} catch (FileNotFoundException e) {
80  					// should not be
81  					wrong.add(filename);
82  					continue;
83  				}
84  				int read = 0;
85  				int offset = 0;
86  				while (read > 0) {
87  					try {
88  						read = inputStream.read(key, offset, len);
89  					} catch (IOException e) {
90  						wrong.add(filename);
91  						read = -2;
92  						break;
93  					}
94  					offset += read;
95  					if (offset < len) {
96  						len -= read;
97  					} else {
98  						break;
99  					}
100 				}
101 				try {
102 					inputStream.close();
103 				} catch (IOException e) {
104 				}
105 				if (read < -1) {
106 					// wrong
107 					continue;
108 				}
109 				String infFilename = filename + INFEXTENSION;
110 				File infFile = new File(infFilename);
111 				inputStream = null;
112 				try {
113 					inputStream = new FileInputStream(infFile);
114 				} catch (FileNotFoundException e) {
115 					// should not be
116 					wrong.add(filename);
117 					continue;
118 				}
119 				KeyObject keyObject;
120 				try {
121 					int keySize = inputStream.read();
122 					String algo = readString(inputStream);
123 					if (algo == null) {
124 						wrong.add(filename);
125 						continue;
126 					}
127 					String instance = readString(inputStream);
128 					if (instance == null) {
129 						wrong.add(filename);
130 						continue;
131 					}
132 					keyObject = new DynamicKeyObject(keySize, algo, instance,
133 							extension);
134 				} catch (IOException e1) {
135 					wrong.add(filename);
136 					continue;
137 				} finally {
138 					try {
139 						inputStream.close();
140 					} catch (IOException e) {
141 					}
142 				}
143 				keyObject.setSecretKey(key);
144 				this.setKey(firstname, keyObject);
145 			} else {
146 				wrong.add(filename);
147 			}
148 		}
149 		this.isInitialized.set(true);
150 		return wrong;
151 	}

153 	/**
154 	 * Specific functions to ease the process of reading the "inf" file
155 	 * 
156 	 * @param inputStream
157 	 * @return the String that should be read
158 	 */
159 	private String readString(FileInputStream inputStream) {
160 		int len;
161 		try {
162 			len = inputStream.read();
163 		} catch (IOException e1) {
164 			return null;
165 		}
166 		byte[] readbyte = new byte[len];
167 		for (int i = 0; i < len; i++) {
168 			try {
169 				readbyte[i] = (byte) inputStream.read();
170 			} catch (IOException e) {
171 				return null;
172 			}
173 		}
174 		return new String(readbyte);
175 	}

177 	@Override
178 	public void saveToFiles(String extension) throws CryptoException,
179 			IOException {
180 		Enumeration<String> names = keysConcurrentHashMap.keys();
181 		while (names.hasMoreElements()) {
182 			String name = names.nextElement();
183 			KeyObject key = keysConcurrentHashMap.get(name);
184 			key.saveSecretKey(new File(name + "." + extension));
185 			FileOutputStream outputStream = new FileOutputStream(new File(name +
186 					"." + extension + INFEXTENSION));
187 			try {
188 				outputStream.write(key.getKeySize());
189 				String algo = key.getAlgorithm();
190 				String instance = key.getInstance();
191 				outputStream.write(algo.length());
192 				outputStream.write(algo.getBytes());
193 				outputStream.write(instance.length());
194 				outputStream.write(instance.getBytes());
195 				outputStream.close();
196 			} finally {
197 				outputStream.close();
198 			}
199 		}
200 	}
