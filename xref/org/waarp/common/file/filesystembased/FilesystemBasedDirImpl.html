117 	 * Finds all files matching a wildcard expression (based on '?', '~' or '*').
118 	 * 
119 	 * @param pathWithWildcard
120 	 *            The wildcard expression with a business path.
121 	 * @return List of String as relative paths matching the wildcard expression. Those files are
122 	 *         tested as valid from business point of view. If Wildcard support is not active, if
123 	 *         the path contains any wildcards, it will throw an error.
124 	 * @throws CommandAbstractException
125 	 */
126 	protected List<String> wildcardFiles(String pathWithWildcard)
127 			throws CommandAbstractException {
128 		List<String> resultPaths = new ArrayList<String>();
129 		// First check if pathWithWildcard contains wildcards
130 		if (!(pathWithWildcard.contains("*") || pathWithWildcard.contains("?") || pathWithWildcard
131 				.contains("~"))) {
132 			// No so simply return the list containing this path after
133 			// validating it
134 			if (getSession().getAuth().isBusinessPathValid(pathWithWildcard)) {
135 				resultPaths.add(pathWithWildcard);
136 			}
137 			return resultPaths;
138 		}
139 		// Do we support Wildcard path
140 		if (!FilesystemBasedDirJdkAbstract.ueApacheCommonsIo) {
141 			throw new Reply553Exception("Wildcards in pathname is not allowed");
142 		}
143 		File wildcardFile;
144 		File rootFile;
145 		if (isAbsoluteWindows(pathWithWildcard)) {
146 			wildcardFile = new File(pathWithWildcard);
147 			rootFile = getCorrespondingRoot(wildcardFile);
148 		} else {
149 			rootFile = new File(((FilesystemBasedAuthImpl) getSession()
150 					.getAuth()).getBaseDirectory());
151 			wildcardFile = new File(rootFile, pathWithWildcard);
152 		}
153 		// Split wildcard path into subdirectories.
154 		List<String> subdirs = new ArrayList<String>();
155 		while (wildcardFile != null) {
156 			File parent = wildcardFile.getParentFile();
157 			if (parent == null) {
158 				subdirs.add(0, wildcardFile.getPath());
159 				break;
160 			}
161 			subdirs.add(0, wildcardFile.getName());
162 			if (parent.equals(rootFile)) {
163 				// End of wildcard path
164 				subdirs.add(0, parent.getPath());
165 				break;
166 			}
167 			wildcardFile = parent;
168 		}
169 		List<File> basedPaths = new ArrayList<File>();
170 		// First set root
171 		basedPaths.add(new File(subdirs.get(0)));
172 		int i = 1;
173 		// For each wilcard subdirectory
174 		while (i < subdirs.size()) {
175 			// Set current filter
176 			FileFilter fileFilter = FilesystemBasedCommonsIo
177 					.getWildcardFileFilter(subdirs.get(i));
178 			List<File> newBasedPaths = new ArrayList<File>();
179 			// Look for matches in all the current search paths
180 			for (File dir : basedPaths) {
181 				if (dir.isDirectory()) {
182 					for (File match : dir.listFiles(fileFilter)) {
183 						newBasedPaths.add(match);
184 					}
185 				}
186 			}
187 			// base Search Path changes now
188 			basedPaths = newBasedPaths;
189 			i++;
190 		}
191 		// Valid each file first
192 		for (File file : basedPaths) {
193 			String relativePath = ((FilesystemBasedAuthImpl) getSession()
194 					.getAuth()).getRelativePath(normalizePath(file
195 					.getAbsolutePath()));
196 			String newpath = this.validatePath(relativePath);
197 			resultPaths.add(newpath);
198 		}
199 		return resultPaths;
200 	}

203 	 * Get the FileInterface from this path, checking first its validity
204 	 * 
205 	 * @param path
206 	 * @return the FileInterface
207 	 * @throws CommandAbstractException
208 	 */
209 	protected File getFileFromPath(String path) throws CommandAbstractException {
210 		String newdir = validatePath(path);
211 		if (isAbsoluteWindows(newdir)) {
212 			return new File(newdir);
213 		}
214 		String truedir = ((FilesystemBasedAuthImpl) getSession().getAuth())
215 				.getAbsolutePath(newdir);
216 		return new File(truedir);
217 	}

220 	 * Get the true file from the path
221 	 * 
222 	 * @param path
223 	 * @return the true File from the path
224 	 * @throws CommandAbstractException
225 	 */
226 	protected File getTrueFile(String path) throws CommandAbstractException {
227 		checkIdentify();
228 		String newpath = consolidatePath(path);
229 		List<String> paths = wildcardFiles(normalizePath(newpath));
230 		if (paths.size() != 1) {
231 			throw new Reply550Exception("File not found: " + paths.size() +
232 					" founds");
233 		}
234 		String extDir = paths.get(0);
235 		extDir = this.validatePath(extDir);
236 		File file = getFileFromPath(extDir);
237 		if (!file.isFile()) {
238 			throw new Reply550Exception("Path is not a file: " + path);
239 		}
240 		return file;
241 	}

254 	public boolean changeDirectory(String path) throws CommandAbstractException {
255 		checkIdentify();
256 		String newpath = consolidatePath(path);
257 		List<String> paths = wildcardFiles(newpath);
258 		if (paths.size() != 1) {
259 			logger.warn("CD error: {}", newpath);
260 			throw new Reply550Exception("Directory not found: " + paths.size() +
261 					" founds");
262 		}
263 		String extDir = paths.get(0);
264 		extDir = this.validatePath(extDir);
265 		if (isDirectory(extDir)) {
266 			currentDir = extDir;
267 			return true;
268 		}
269 		throw new Reply550Exception("Directory not found: " + extDir);
270 	}

286 	public String mkdir(String directory) throws CommandAbstractException {
287 		checkIdentify();
288 		String newdirectory = consolidatePath(directory);
289 		File dir = new File(newdirectory);
290 		String parent = dir.getParentFile().getPath();
291 		List<String> paths = wildcardFiles(normalizePath(parent));
292 		if (paths.size() != 1) {
293 			throw new Reply550Exception("Base Directory not found: " +
294 					paths.size() + " founds");
295 		}
296 		String newDir = paths.get(0) + SEPARATOR + dir.getName();
297 		newDir = this.validatePath(newDir);
298 		File newdir = getFileFromPath(newDir);
299 		if (newdir.mkdir()) {
300 			return newDir;
301 		}
302 		throw new Reply550Exception("Cannot create directory " + newDir);
303 	}

344 	 * Return the Modification time for the File
345 	 * 
346 	 * @param file
347 	 * @return the Modification time as a String YYYYMMDDHHMMSS.sss
348 	 */
349 	protected String getModificationTime(File file) {
350 		long mstime = file.lastModified();
351 		Calendar calendar = Calendar.getInstance();
352 		calendar.setTimeInMillis(mstime);
353 		int year = calendar.get(Calendar.YEAR);
354 		int month = calendar.get(Calendar.MONTH) + 1;
355 		int day = calendar.get(Calendar.DAY_OF_MONTH);
356 		int hour = calendar.get(Calendar.HOUR_OF_DAY);
357 		int minute = calendar.get(Calendar.MINUTE);
358 		int second = calendar.get(Calendar.SECOND);
359 		int ms = calendar.get(Calendar.MILLISECOND);
360 		StringBuilder sb = new StringBuilder(18);
361 		sb.append(year);
362 		if (month < 10) {
363 			sb.append(0);
364 		}
365 		sb.append(month);
366 		if (day < 10) {
367 			sb.append(0);
368 		}
369 		sb.append(day);
370 		if (hour < 10) {
371 			sb.append(0);
372 		}
373 		sb.append(hour);
374 		if (minute < 10) {
375 			sb.append(0);
376 		}
377 		sb.append(minute);
378 		if (second < 10) {
379 			sb.append(0);
380 		}
381 		sb.append(second);
382 		sb.append('.');
383 		if (ms < 10) {
384 			sb.append(0);
385 		}
386 		if (ms < 100) {
387 			sb.append(0);
388 		}
389 		sb.append(ms);
390 		return sb.toString();
391 	}

393 	public List<String> list(String path) throws CommandAbstractException {
394 		checkIdentify();
395 		// First get all base directories
396 		String newpath = path;
397 		if (newpath == null || newpath.length() == 0) {
398 			newpath = currentDir;
399 		}
400 		if (newpath.startsWith("-a") || newpath.startsWith("-A")) {
401 			String[] args = newpath.split(" ");
402 			if (args.length > 1) {
403 				newpath = args[1];
404 			} else {
405 				newpath = currentDir;
406 			}
407 		}
408 		newpath = consolidatePath(newpath);
409 		List<String> paths = wildcardFiles(newpath);
410 		if (paths.isEmpty()) {
411 			throw new Reply550Exception("No files found");
412 		}
413 		// Now if they are directories, list inside them
414 		List<String> newPaths = new ArrayList<String>();
415 		for (String file : paths) {
416 			File dir = getFileFromPath(file);
417 			if (dir.exists()) {
418 				if (dir.isDirectory()) {
419 					String[] files = dir.list();
420 					for (String finalFile : files) {
421 						String relativePath = ((FilesystemBasedAuthImpl) getSession()
422 								.getAuth()).getRelativePath(finalFile);
423 						newPaths.add(relativePath);
424 					}
425 				} else {
426 					newPaths.add(file);
427 				}
428 			}
429 		}
430 		return newPaths;
431 	}

433 	public List<String> listFull(String path, boolean lsFormat)
434 			throws CommandAbstractException {
435 		checkIdentify();
436 		boolean listAllFiles = false;
437 		String newpath = path;
438 		if (newpath == null || newpath.length() == 0) {
439 			newpath = currentDir;
440 		}
441 		if (newpath.startsWith("-a") || newpath.startsWith("-A")) {
442 			String[] args = newpath.split(" ");
443 			if (args.length > 1) {
444 				newpath = args[1];
445 			} else {
446 				newpath = currentDir;
447 			}
448 			listAllFiles = true;
449 		}
450 		newpath = consolidatePath(newpath);
451 		// First get all base directories
452 		List<String> paths = wildcardFiles(newpath);
453 		if (paths.isEmpty()) {
454 			throw new Reply550Exception("No files found");
455 		}
456 		// Now if they are directories, list inside them
457 		List<String> newPaths = new ArrayList<String>();
458 		for (String file : paths) {
459 			File dir = getFileFromPath(file);
460 			if (dir.exists()) {
461 				if (dir.isDirectory()) {
462 					File[] files = dir.listFiles();
463 					for (File finalFile : files) {
464 						if (lsFormat) {
465 							newPaths.add(lsInfo(finalFile));
466 						} else {
467 							newPaths.add(mlsxInfo(finalFile));
468 						}
469 					}
470 				} else {
471 					if (lsFormat) {
472 						newPaths.add(lsInfo(dir));
473 					} else {
474 						newPaths.add(mlsxInfo(dir));
475 					}
476 				}
477 			}
478 		}
479 		if (listAllFiles) {
480 			File dir = new File(getFileFromPath(newpath), SEPARATOR + "..");
481 			if (lsFormat) {
482 				newPaths.add(lsInfo(dir));
483 			} else {
484 				newPaths.add(mlsxInfo(dir));
485 			}
486 		}
487 		return newPaths;
488 	}

491 			throws CommandAbstractException {
492 		checkIdentify();
493 		String newpath = consolidatePath(path);
494 		List<String> paths = wildcardFiles(normalizePath(newpath));
495 		if (paths.size() != 1) {
496 			throw new Reply550Exception("No files found " + paths.size() +
497 					" founds");
498 		}
499 		File file = getFileFromPath(paths.get(0));
500 		if (file.exists()) {
501 			if (lsFormat) {
502 				return "Listing of \"" + paths.get(0) + "\"\n" + lsInfo(file) +
503 						"\nEnd of listing";
504 			}
505 			return "Listing of \"" + paths.get(0) + "\"\n" + mlsxInfo(file) +
506 					"\nEnd of listing";
507 		}
508 		return "No file with name \"" + path + "\"";
509 	}

521 	/**
522 	 * 
523 	 * @param file
524 	 * @return the ls format information
525 	 */
526 	protected String lsInfo(File file) {
527 		// Unix FileInterface type,permissions,hard
528 		// link(?),owner(?),group(?),size,date
529 		// and filename
530 		StringBuilder builder = new StringBuilder();
531 		builder.append((file.isDirectory() ? 'd' : '-'));
532 		builder.append((file.canRead() ? 'r' : '-'));
533 		builder.append((file.canWrite() ? 'w' : '-'));
534 		if (filesystemBasedFtpDirJdk != null)
535 			builder.append(filesystemBasedFtpDirJdk.canExecute(file) ? 'x' : '-');
536 		else
537 			builder.append('-');
538 		// Group and others not supported
539 		builder.append("---");
540 		builder.append("---");
541 		builder.append(' ');
542 		builder.append("1 ");// hard link ?
543 		builder.append("anybody\t");// owner ?
544 		builder.append("anygroup\t");// group ?
545 		builder.append(file.length());// size
546 		builder.append('\t');
547 		long lastmod = file.lastModified();
548 		String fmt = null;
549 		// It seems Full Time is not recognized by some FTP client
550 		/*
551 		 * if(isFullTime()) { fmt = "EEE MMM dd HH:mm:ss yyyy"; } else {
552 		 */
553 		long currentTime = System.currentTimeMillis();
554 		if (currentTime > lastmod + 6L * 30L * 24L * 60L * 60L * 1000L // Old.
555 				|| currentTime < lastmod - 60L * 60L * 1000L) { // In the
556 			// future.
557 			// The file is fairly old or in the future.
558 			// POSIX says the cutoff is 6 months old;
559 			// approximate this by 6*30 days.
560 			// Allow a 1 hour slop factor for what is considered "the future",
561 			// to allow for NFS server/client clock disagreement.
562 			// Show the year instead of the time of day.
563 			fmt = "MMM dd  yyyy";
564 		} else {
565 			fmt = "MMM dd HH:mm";
566 		}
567 		/* } */
568 		SimpleDateFormat dateFormat = (SimpleDateFormat) DateFormat
569 				.getDateTimeInstance(DateFormat.LONG, DateFormat.LONG,
570 						Locale.ENGLISH);
571 		dateFormat.applyPattern(fmt);
572 		builder.append(dateFormat.format(new Date(lastmod)));// date
573 		builder.append('\t');
574 		builder.append(file.getName());
575 		return builder.toString();
576 	}

578 	/**
579 	 * 
580 	 * @param file
581 	 * @return the MLSx information: ' Fact=facts;...; filename'
582 	 */
583 	protected String mlsxInfo(File file) {
584 		// don't have create, unique, lang, media-type, charset
585 		StringBuilder builder = new StringBuilder(" ");
586 		if (getOptsMLSx().getOptsSize() == 1) {
587 			builder.append("Size=");
588 			builder.append(file.length());
589 			builder.append(';');
590 		}
591 		if (getOptsMLSx().getOptsModify() == 1) {
592 			builder.append("Modify=");
593 			builder.append(this.getModificationTime(file));
594 			builder.append(';');
595 		}
596 		if (getOptsMLSx().getOptsType() == 1) {
597 			builder.append("Type=");
598 			try {
599 				if (getFileFromPath(currentDir).equals(file)) {
600 					builder.append("cdir");
601 				} else {
602 					if (file.isDirectory()) {
603 						builder.append("dir");
604 					} else {
605 						builder.append("file");
606 					}
607 				}
608 			} catch (CommandAbstractException e) {
609 				if (file.isDirectory()) {
610 					builder.append("dir");
611 				} else {
612 					builder.append("file");
613 				}
614 			}
615 			builder.append(';');
616 		}
617 		if (getOptsMLSx().getOptsPerm() == 1) {
618 			builder.append("Perm=");
619 			if (file.isFile()) {
620 				if (file.canWrite()) {
621 					builder.append('a');
622 					builder.append('d');
623 					builder.append('f');
624 					builder.append('w');
625 				}
626 				if (file.canRead()) {
627 					builder.append('r');
628 				}
629 			} else {
630 				// Directory
631 				if (file.canWrite()) {
632 					builder.append('c');
633 					try {
634 						if (this.validatePath(file) != null) {
635 							builder.append('d');
636 							builder.append('m');
637 							builder.append('p');
638 						}
639 					} catch (CommandAbstractException e) {
640 					}
641 				}
642 				if (file.canRead()) {
