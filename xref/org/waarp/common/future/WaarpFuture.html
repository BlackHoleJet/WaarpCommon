110 	/**
111 	 * Rethrows the exception that caused this future fail if this future is complete and failed.
112 	 */
113 	public WaarpFuture rethrowIfFailed() throws Exception {
114 		if (!isDone()) {
115 			return this;
116 		}
117 
118 		Throwable cause = getCause();
119 		if (cause == null) {
120 			return this;
121 		}
122 
123 		if (cause instanceof Exception) {
124 			throw (Exception) cause;
125 		}
126 
127 		if (cause instanceof Error) {
128 			throw (Error) cause;
129 		}
130 
131 		throw new RuntimeException(cause);
132 	}

135 	 * Waits for this future to be completed.
136 	 * 
137 	 * @return The WaarpFuture
138 	 * 
139 	 * @throws InterruptedException
140 	 *             if the current thread was interrupted
141 	 */
142 	public WaarpFuture await() throws InterruptedException {
143 		if (Thread.interrupted()) {
144 			throw new InterruptedException();
145 		}
146 
147 		synchronized (this) {
148 			while (!done) {
149 				waiters++;
150 				try {
151 					this.wait();
152 				} finally {
153 					waiters--;
154 				}
155 			}
156 		}
157 		return this;
158 	}

251 	private boolean await0(long timeoutNanos, boolean interruptable)
252 			throws InterruptedException {
253 		if (interruptable && Thread.interrupted()) {
254 			throw new InterruptedException();
255 		}
256 
257 		long startTime = timeoutNanos <= 0 ? 0 : System.nanoTime();
258 		long waitTime = timeoutNanos;
259 		boolean interrupted = false;
260 
261 		try {
262 			synchronized (this) {
263 				if (done) {
264 					return done;
265 				} else if (waitTime <= 0) {
266 					return done;
267 				}
268 
269 				waiters++;
270 				try {
271 					for (;;) {
272 						try {
273 							this.wait(waitTime / 1000000,
274 									(int) (waitTime % 1000000));
275 						} catch (InterruptedException e) {
276 							if (interruptable) {
277 								throw e;
278 							} else {
279 								interrupted = true;
280 							}
281 						}
282 
283 						if (done) {
284 							return true;
285 						}
286 						waitTime = timeoutNanos - (System.nanoTime() - startTime);
287 						if (waitTime <= 0) {
288 							return done;
289 						}
290 					}
291 				} finally {
292 					waiters--;
293 				}
294 			}
295 		} finally {
296 			if (interrupted) {
297 				Thread.currentThread().interrupt();
298 			}
299 		}
300 	}

302 	/**
303 	 * Marks this future as a success and notifies all listeners.
304 	 * 
305 	 * @return {@code true} if and only if successfully marked this future as a success. Otherwise
306 	 *         {@code false} because this future is already marked as either a success or a failure.
307 	 */
308 	public boolean setSuccess() {
309 		synchronized (this) {
310 			// Allow only once.
311 			if (done) {
312 				return false;
313 			}
314 
315 			done = true;
316 			if (waiters > 0) {
317 				notifyAll();
318 			}
319 		}
320 		return true;
321 	}
322 
323 	/**
324 	 * Marks this future as a failure and notifies all listeners.
325 	 * 
326 	 * @param cause
327 	 * @return {@code true} if and only if successfully marked this future as a failure. Otherwise
328 	 *         {@code false} because this future is already marked as either a success or a failure.
329 	 */
330 	public boolean setFailure(Throwable cause) {
331 		synchronized (this) {
332 			// Allow only once.
333 			if (done) {
334 				return false;
335 			}
336 
337 			this.cause = cause;
338 			done = true;
339 			if (waiters > 0) {
340 				notifyAll();
341 			}
342 		}
343 		return true;
344 	}

346 	/**
347 	 * Cancels the operation associated with this future and notifies all listeners if canceled
348 	 * successfully.
349 	 * 
350 	 * @return {@code true} if and only if the operation has been canceled. {@code false} if the
351 	 *         operation can't be canceled or is already completed.
352 	 */
353 	public boolean cancel() {
354 		if (!cancellable) {
355 			return false;
356 		}
357 		synchronized (this) {
358 			// Allow only once.
359 			if (done) {
360 				return false;
361 			}
362 
363 			cause = CANCELLED;
364 			done = true;
365 			if (waiters > 0) {
366 				notifyAll();
367 			}
368 		}
369 		return true;
370 	}
