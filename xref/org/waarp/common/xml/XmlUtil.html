36  /**
37   * XML utility that handles simple cases as:<br>
38   * <ul>
39   * <li>XPath as /x/y/z for a node and referring to root of Document</li>
40   * <li>XPath as x/y/z for a node and referring to current referenced node</li>
41   * <li>Any XPath can be a singleton (unique node referenced by the XPath) or multiple nodes (same
42   * XPath)</li>
43   * <li>Only Element as Node: no Attribute or any other type within XML</li>
44   * <li>Any other path is not supported: //x /x@y ./ ../</li>
45   * <li>Supports special SubXml tree as element (singleton or multiple)</li>
46   * </ul>
47   * 
48   * @author Frederic Bregier
49   * 
50   */
51  public class XmlUtil {

148 		}
149 		OutputFormat format = OutputFormat.createPrettyPrint();
150 		format.setEncoding(WaarpStringUtils.UTF8.name());
151 		XMLWriter writer = new XMLWriter(new FileWriter(file), format);
152 		writer.write(document);
153 		writer.flush();
154 		writer.close();
155 	}

216 	 * element
217 	 * 
218 	 * @param ref
219 	 * @param path
220 	 * @return the new added or already existing element
221 	 */
222 	static public Element addOrGetElement(Element ref, String path) {
223 		String[] pathes = path.split("/");
224 		Element current = ref;
225 		for (String nodename : pathes) {
226 			if (! nodename.isEmpty()) {
227 				Element exist = current.element(nodename);
228 				if (exist == null) {
229 					current = current.addElement(nodename);
230 				} else {
231 					current = exist;
232 				}
233 			}
234 		}
235 		return current;
236 	}

287 	 */
288 	static public Element getParentElement(Element ref, String path)
289 			throws DocumentException {
290 		String npath = path;
291 		while (npath.charAt(0) == '/') {// startsWith("/")) {
292 			npath = npath.substring(1);
293 		}
294 		Element current = (Element) ref.selectSingleNode(npath);
295 		if (current == null) {
296 			throw new DocumentException("Node not found: " + path);
297 		}
298 		return current.getParent();
299 	}

359 	 * Add or Get (if already existing) an element given by the path relative to the document
360 	 * 
361 	 * @param doc
362 	 * @param path
363 	 * @return the new added or already existing element
364 	 */
365 	static public Element addOrGetElement(Document doc, String path) {
366 		String[] pathes = path.split("/");
367 		int rank = 0;
368 		for (rank = 0; rank < pathes.length; rank++) {
369 			if (! pathes[rank].isEmpty()) {
370 				break; // found
371 			}
372 		}
373 		if (rank >= pathes.length) {
374 			return null; // Should not be !
375 		}
376 		Element current = (Element) doc.selectSingleNode(pathes[rank]);
377 		if (current == null) {
378 			current = doc.addElement(pathes[rank]);
379 		}
380 		for (int i = rank + 1; i < pathes.length; i++) {
381 			String nodename = pathes[i];
382 			if (! nodename.isEmpty()) {
383 				Element exist = current.element(nodename);
384 				if (exist == null) {
385 					current = current.addElement(nodename);
386 				} else {
387 					current = exist;
388 				}
389 			}
390 		}
391 		return current;
392 	}

503 	/**
504 	 * Create the XmlValues from the XmlDevls and the Document
505 	 * 
506 	 * @param doc
507 	 * @param decls
508 	 * @return XmlValues
509 	 */
510 	static public XmlValue[] read(Document doc, XmlDecl[] decls) {
511 		XmlValue[] values = null;
512 		int len = decls.length;
513 		values = new XmlValue[len];
514 		for (int i = 0; i < len; i++) {
515 			XmlValue value = new XmlValue(decls[i]);
516 			values[i] = value;
517 			if (decls[i].isSubXml()) {
518 				if (decls[i].isMultiple()) {
519 					List<Element> elts;
520 					try {
521 						elts = getElementMultiple(doc, decls[i].getXmlPath());
522 					} catch (DocumentException e) {
523 						continue;
524 					}
525 					for (Element element : elts) {
526 						XmlValue[] newValue = read(element,
527 								decls[i].getSubXml());
528 						if (newValue == null) {
529 							continue;
530 						}
531 						try {
532 							value.addValue(newValue);
533 						} catch (InvalidObjectException e) {
534 							continue;
535 						}
536 					}
537 				} else {
538 					Element element;
539 					try {
540 						element = getElement(doc, decls[i].getXmlPath());
541 					} catch (DocumentException e) {
542 						continue;
543 					}
544 					XmlValue[] newValue = read(element, decls[i].getSubXml());
545 					if (newValue == null) {
546 						continue;
547 					}
548 					try {
549 						value.setValue(newValue);
550 					} catch (InvalidObjectException e) {
551 						continue;
552 					}
553 				}
554 			} else if (decls[i].isMultiple()) {
555 				List<Element> elts;
556 				try {
557 					elts = getElementMultiple(doc, decls[i].getXmlPath());
558 				} catch (DocumentException e) {
559 					continue;
560 				}
561 				for (Element element : elts) {
562 					String svalue = element.getText();
563 					try {
564 						value.addFromString(svalue);
565 					} catch (InvalidObjectException e) {
566 						continue;
567 					} catch (InvalidArgumentException e) {
568 						continue;
569 					}
570 				}
571 			} else {
572 				Element element;
573 				try {
574 					element = getElement(doc, decls[i].getXmlPath());
575 				} catch (DocumentException e) {
576 					continue;
577 				}
578 				String svalue = element.getText();
579 				try {
580 					value.setFromString(svalue);
581 				} catch (InvalidArgumentException e) {
582 					continue;
583 				}
584 			}
585 		}
586 		return values;
587 	}

675 	/**
676 	 * Add all nodes from XmlValues into Document
677 	 * 
678 	 * @param doc
679 	 * @param values
680 	 */
681 	@SuppressWarnings("unchecked")
682 	public static void write(Document doc, XmlValue[] values) {
683 		int len = values.length;
684 		for (int i = 0; i < len; i++) {
685 			if (values[i] != null) {
686 				if (values[i].isSubXml()) {
687 					if (values[i].isMultiple()) {
688 						List<XmlValue[]> list = (List<XmlValue[]>) values[i]
689 								.getList();
690 						for (XmlValue[] object : list) {
691 							Element ref = addAndGetElementMultiple(doc,
692 									values[i].getXmlPath());
693 							write(ref, object);
694 						}
695 					} else {
696 						Element ref = addOrGetElement(doc,
697 								values[i].getXmlPath());
698 						write(ref, values[i].getSubXml());
699 					}
700 				} else if (values[i].isMultiple()) {
701 					List<?> list = values[i].getList();
702 					for (Object object : list) {
703 						addAndSetElementMultiple(doc, values[i].getXmlPath(),
704 								object.toString());
705 					}
706 				} else {
707 					addOrSetElement(doc, values[i].getXmlPath(),
708 							values[i].getIntoString());
709 				}
710 			}
711 		}
712 	}
